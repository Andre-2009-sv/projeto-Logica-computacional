Calculadora:
INÍCIO
    DECLARAR operador como CARACTERE
    DECLARAR num1, num2, resultado como REAL

    EXIBIR "Calculadora"

    EXIBIR "Digite o primeiro número: "
    LER num1

    EXIBIR "Digite o segundo número: "
    LER num2

    EXIBIR "Digite o operador (+, -, *, /): "
    LER operador

    SELECIONAR operador
        CASO '+':
            resultado = num1 + num2
            EXIBIR "Resultado: ", resultado
            SAIR
        CASO '-':
            resultado = num1 - num2
            EXIBIR "Resultado: ", resultado
            SAIR
        CASO '*':
            resultado = num1 * num2
            EXIBIR "Resultado: ", resultado
            SAIR
        CASO '/':
            SE num2 é DIFERENTE DE 0 ENTÃO
                resultado = num1 / num2
                EXIBIR "Resultado: ", resultado
            SENÃO
                EXIBIR "Divisão por zero não é permitida."
            FIM SE
            SAIR
        CASO CONTRÁRIO:
            EXIBIR "Operador inválido. Por favor, use +, -, *, ou /."
    FIM SELECIONAR

    RETORNAR 0
FIM
